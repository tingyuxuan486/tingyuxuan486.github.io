若依分页问题

1、必须要在查询列表的业务前加startPage();

2、如果多次查询或者查询需要做筛选导致出现两个list（其中第二个list会丢掉分页信息）

3、如果先分页，再过滤会导致可能某一页出现实际pagesize少于需要显示的pagesize

第一种（如果是需要先查询所有的，然后过滤掉一部分，再拿到分页信息）

@PreAuthorize("@ss.hasPermi('system:achievement:list')")
@GetMapping("/list")
public TableDataInfo list(NekndResearchAchievement nekndResearchAchievement,@RequestParam(required = false) Integer pageNum, @RequestParam(required = false) Integer pageSize)
{
List<NekndResearchAchievement> list = nekndResearchAchievementService.selectNekndResearchAchievementList(nekndResearchAchievement);
            //如果是教育端，则只显示当前部门且已审核的
            List<NekndResearchAchievement> achievementList = list.stream()
                    .filter(item -> item.getDeptId() == Integer.parseInt(user.getDeptId() + ""))
                    .collect(Collectors.toList());
            //自定义的分页工具
            return CustomPageUtils.paginateList(achievementList, pageNum, pageSize);
}

public class CustomPageUtils {
    /**
     * 对给定的列表进行分页处理。
     *
     * @param <T> 列表元素的类型
     * @param list    要分页的原始列表
     * @param pageNum 页码，从1开始
     * @param pageSize 每页显示的记录数
     * @return 包含分页后数据和总记录数的TableDataInfo对象
     */
    public static <T> TableDataInfo paginateList(List<T> list, Integer pageNum, Integer pageSize) {
        if (list == null || list.isEmpty()) {
            return new TableDataInfo(new ArrayList<>(), 0){
                {
                    setCode(HttpStatus.SUCCESS);
                    setMsg("获取列表数据成功！");
                }
            }; // 如果没有数据，则返回一个空的TableDataInfo对象
        }
        TableDataInfo rspData = null;
        if (pageNum == null || pageSize == null) {
            rspData = new TableDataInfo(list, list.size()); // 如果页码或每页显示的记录数没有传入，则返回原始列表和总记录数
            rspData.setCode(HttpStatus.SUCCESS);
            rspData.setMsg("获取全部列表数据成功！");
            return rspData;
        }
        int size = list.size();
        int pageStart = (pageNum - 1) * pageSize;
        int pageEnd = Math.min(size, pageNum * pageSize);

        List<T> paginatedList = new ArrayList<>(pageEnd - pageStart);
        if (pageStart < size) {
            paginatedList.addAll(list.subList(pageStart, pageEnd));
        }

        rspData = new TableDataInfo(paginatedList, size);// 返回分页后的列表和总记录数
        rspData.setCode(HttpStatus.SUCCESS);
        rspData.setMsg("获取列表数据成功！");
        return rspData;
    }
